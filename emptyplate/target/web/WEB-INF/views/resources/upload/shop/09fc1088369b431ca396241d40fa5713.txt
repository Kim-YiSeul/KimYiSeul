CREATE TABLE WEATHER_IMSI 
(	AREA CHAR(3) NOT NULL, 
	AREA_NAME VARCHAR2(200), 
	CUR_DATE DATE, 
	TEMPERATURE NUMBER(5,1), 
	RAINFALL NUMBER(5,1), 
	WIND_SPEED NUMBER(5,1), 
	WIND_DIRECTION NUMBER(5,0), 
	HUMIDITY NUMBER(5,0), 
	AIR_BAROMETRIC NUMBER(7,1), 
	SUN_LIGHT NUMBER(5,1), 
	SUN_LIGHT_STRONG NUMBER(7,2), 
	DEEP_SNOW NUMBER(5,1), 
	CLOUD NUMBER(5,0), 
	VISIBILITY NUMBER(5,0), 
	GROUND_STATUS NUMBER(5,0), 
	GROUND_TEMPER NUMBER(5,1)
);

   COMMENT ON COLUMN WEATHER_IMSI.AREA IS '지역코드';
   COMMENT ON COLUMN WEATHER_IMSI.AREA_NAME IS '지역명';
   COMMENT ON COLUMN WEATHER_IMSI.CUR_DATE IS '날씨날짜';
   COMMENT ON COLUMN WEATHER_IMSI.TEMPERATURE IS '기온';
   COMMENT ON COLUMN WEATHER_IMSI.RAINFALL IS '강수량';
   COMMENT ON COLUMN WEATHER_IMSI.WIND_SPEED IS '풍속';
   COMMENT ON COLUMN WEATHER_IMSI.WIND_DIRECTION IS '풍향(16방위)';
   COMMENT ON COLUMN WEATHER_IMSI.HUMIDITY IS '습도';
   COMMENT ON COLUMN WEATHER_IMSI.AIR_BAROMETRIC IS '현지기압(hPa)';
   COMMENT ON COLUMN WEATHER_IMSI.SUN_LIGHT IS '일조(hr)';
   COMMENT ON COLUMN WEATHER_IMSI.SUN_LIGHT_STRONG IS '일사(MJ/m2)';
   COMMENT ON COLUMN WEATHER_IMSI.DEEP_SNOW IS '적설(cm)';
   COMMENT ON COLUMN WEATHER_IMSI.CLOUD IS '전운량(10분위)';
   COMMENT ON COLUMN WEATHER_IMSI.VISIBILITY IS '시정(10m)';
   COMMENT ON COLUMN WEATHER_IMSI.GROUND_STATUS IS '지면상태(지면상태코드)';
   COMMENT ON COLUMN WEATHER_IMSI.GROUND_TEMPER IS '지면온도(C)';
   COMMENT ON TABLE WEATHER_IMSI  IS '일기예보';
